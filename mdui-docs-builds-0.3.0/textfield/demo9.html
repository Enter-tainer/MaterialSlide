var $$ = mdui.JQ;

var $textfield = $$('#textfield');
var $input = $$('#input');
var $error = $$('#error');
var hasError = false;

$$('#valid').on('click', function () {
  var value = $input.val();

  if (value.length < 6) {
    $error.text('用户名不能小于 6 位');
    hasError = true;
  } else if (value.length > 20) {
    $error.text('用户名不能超过 20 位');
    hasError = true;
  } else {
    hasError = false;
  }

  if (hasError) {
    $textfield.addClass('mdui-textfield-invalid');
  } else {
    $textfield.removeClass('mdui-textfield-invalid');
  }
});
